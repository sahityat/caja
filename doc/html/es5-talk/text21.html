Tamper proof points

function Point(x, y) {

  return Point.build({

    x: {value: +x, enumerable: true},

    y: {value: +y, enumerable: true}

});

}

var pt = new Point(3,5);

pt.x = 'foo'; // fails, as it should

new Point(3,5) instanceof Point === true

Point(3,5) instanceof Point === true
