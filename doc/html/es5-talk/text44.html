But static scoping isn’t lexical scoping

function sqFnList(a) {

   var b = [];

   for (var i = 0; i < a.length; i++) {

     var sq = a[i] * a[i];

     b[i] = function() { return sq; };

   }

   return b;

}

sqFnList([3,4,5])[0]() //Returns 25. Why?
