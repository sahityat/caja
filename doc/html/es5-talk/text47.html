Lexical Scoping just missed ES5 train. How to cope?

function sqFnList(a) {

   var b = []; var sq;

   for (var i = 0; i < a.length; i++) {

     sq = a[i] * a[i];

     b[i] = function() { return sq; };

   }

   return b;

}

sqFnList([3,4,5])[0]() //Returns 25. Why?

Because of var hoisting.
